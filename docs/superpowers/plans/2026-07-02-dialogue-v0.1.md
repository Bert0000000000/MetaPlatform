## 1. Vision & Scope

对话层是 MetaPlatform 的自然语言交互入口。用户通过对话界面与平台交互，系统将自然语言解析为结构化 Intent，路由到对应的 Handler 执行业务逻辑，再生成自然语言回复。v0.1 聚焦于核心对话管道：消息收发、意图识别与执行、上下文管理。

### 1.1 核心能力

| 能力 | 说明 |
|------|------|
| Conversation 聚合根 | 会话生命周期管理：创建、归档、导出 |
| Message 实体 | 用户/系统/助手消息，支持富文本 |
| Intent 体系 | Intent 定义 → 注册 → 路由 → Handler 四层抽象 |
| 内置 Intent Handler | greeting、help、对象 CRUD（create/query/update/delete） |
| NL Parser | 基于 LLM 的意图提取（调用 AI Substrate LLM Gateway） |
| Response Generator | 模板 + NL 混合回复生成 |
| Context Manager | 桥接 AI Substrate ContextStore，维护对话上下文滑动窗口 |
| MessageService | 核心管道：用户消息 → 意图路由 → 执行 → 响应 |
| REST API | 会话管理 + 消息收发 + 意图查询 + 会话导出 |
| 持久化 | PostgreSQL + Redis + Flyway |

### 1.2 Non-Goals (v0.1)

- 多轮复杂推理链 (v0.2)
- 语音对话 (v0.2)
- 多模态消息（图片/文件）(v0.2)
- 自定义对话流编排 (v0.2)
- 多租户隔离
- 对话分析/统计仪表盘

## 3. Domain Model

### 3.1 Conversation (Aggregate Root)

```java
@Entity
@Table(name = "conversation")
public class Conversation {
    @Id @GeneratedValue(strategy = GenerationType.IDENTITY)
    private Long id;

    @Column(nullable = false)
    private String title;                    // 会话标题（自动生成或用户指定）

    @Enumerated(EnumType.STRING)
    @Column(nullable = false)
    private ConversationStatus status;       // ACTIVE | ARCHIVED | DELETED

    @Column(nullable = false)
    private String userId;                   // 所属用户

    @Column(nullable = false)
    private String channel;                  // WEB | API | BOT

    private String contextKey;               // 关联 AI Substrate ContextStore 的 key

    private LocalDateTime createdAt;
    private LocalDateTime updatedAt;
    private LocalDateTime archivedAt;
}

public enum ConversationStatus {
    ACTIVE, ARCHIVED, DELETED
}
```

### 3.2 Message (Entity)

```java
@Entity
@Table(name = "message")
public class Message {
    @Id @GeneratedValue(strategy = GenerationType.IDENTITY)
    private Long id;

    @ManyToOne(fetch = FetchType.LAZY)
    @JoinColumn(name = "conversation_id", nullable = false)
    private Conversation conversation;

    @Enumerated(EnumType.STRING)
    @Column(nullable = false)
    private MessageRole role;               // USER | ASSISTANT | SYSTEM

    @Column(columnDefinition = "TEXT", nullable = false)
    private String content;                 // 消息正文

    @Column(columnDefinition = "TEXT")
    private String parsedIntent;            // 解析出的意图 JSON（可选）

    @Column(columnDefinition = "TEXT")
    private String executionResult;         // 执行结果 JSON（可选）

    private Integer tokenCount;             // token 计数

    private LocalDateTime createdAt;
}

public enum MessageRole {
    USER, ASSISTANT, SYSTEM
}
```

```java
package com.metaplatform.dialogue.domain;

import java.util.Map;

/**
 * LLM 解析后的意图实例
 */
public record ParsedIntent(
    String intentId,
    double confidence,
    Map<String, Object> slots
) {
    public ParsedIntent {
        if (intentId == null || intentId.isBlank()) {
            throw new IllegalArgumentException("intentId must not be blank");
        }
        if (confidence < 0 || confidence > 1) {
            throw new IllegalArgumentException("confidence must be between 0 and 1");
        }
    }

    /**
     * 是否为高置信度（阈值可配置）
     */
    public boolean isHighConfidence(double threshold) {
        return confidence >= threshold;
    }

    /**
     * 获取槽值，带类型转换
     */
    @SuppressWarnings("unchecked")
    public <T> T getSlot(String name, Class<T> type) {
        if (slots == null) return null;
        Object val = slots.get(name);
        if (val == null) return null;
        if (type.isInstance(val)) return (T) val;
        return null;
    }
}
```

public class IntentRouterImpl implements IntentRouter {

    private static final Logger log = LoggerFactory.getLogger(IntentRouterImpl.class);

    private final IntentRegistry registry;
    private final ApplicationContext applicationContext;

    public IntentRouterImpl(IntentRegistry registry, ApplicationContext applicationContext) {
        this.registry = registry;
        this.applicationContext = applicationContext;
    }

    @Override
    public IntentHandleResult route(ParsedIntent parsedIntent) {
        log.info("Routing intent: {} (confidence={})", parsedIntent.intentId(), parsedIntent.confidence());

        var definition = registry.findByIntentId(parsedIntent.intentId());
        if (definition.isEmpty()) {
            log.warn("No handler registered for intent: {}", parsedIntent.intentId());
            return IntentHandleResult.failure("未识别的意图: " + parsedIntent.intentId());
        }

        IntentDefinition def = definition.get();
        if (!def.isEnabled()) {
            return IntentHandleResult.failure("该意图已被禁用: " + parsedIntent.intentId());
        }

        try {
            IntentHandler handler = applicationContext.getBean(def.getHandlerBeanName(), IntentHandler.class);
            Map<String, Object> context = Map.of();  // v0.1 简化上下文
            return handler.handle(parsedIntent, context);
        } catch (Exception e) {
            log.error("Error handling intent: {}", parsedIntent.intentId(), e);
            return IntentHandleResult.failure("处理意图时发生错误: " + e.getMessage());
        }
    }
}

package com.metaplatform.dialogue.nl;

import com.fasterxml.jackson.core.type.TypeReference;
import com.fasterxml.jackson.databind.ObjectMapper;
import com.metaplatform.dialogue.domain.IntentDefinition;
import com.metaplatform.dialogue.domain.ParsedIntent;
import com.metaplatform.dialogue.intent.IntentRegistry;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.beans.factory.annotation.Value;
import org.springframework.stereotype.Component;
import org.springframework.web.reactive.function.client.WebClient;

import java.util.*;

@Component
public class LlmNlParser implements NlParser {

    private static final Logger log = LoggerFactory.getLogger(LlmNlParser.class);

    private final WebClient aiSubstrateClient;
    private final IntentRegistry intentRegistry;
    private final ObjectMapper objectMapper;

    @Value("${ai-substrate.base-url:http://localhost:8083}")
    private String aiSubstrateBaseUrl;

    @Value("${nl-parser.confidence-threshold:0.7}")
    private double confidenceThreshold;

    public LlmNlParser(WebClient.Builder webClientBuilder,
                       IntentRegistry intentRegistry,
                       ObjectMapper objectMapper) {
        this.aiSubstrateClient = webClientBuilder.build();
        this.intentRegistry = intentRegistry;
        this.objectMapper = objectMapper;
    }

    @Override
    public List<ParsedIntent> parse(String userMessage, String conversationContext) {
        String systemPrompt = buildSystemPrompt();

        Map<String, Object> requestBody = Map.of(
                "model", "smart",
                "messages", List.of(
                        Map.of("role", "system", "content", systemPrompt),
                        Map.of("role", "user", "content", userMessage)
                ),
                "temperature", 0.1,
                "maxTokens", 500
        );

        try {
            Map response = aiSubstrateClient.post()
                    .uri(aiSubstrateBaseUrl + "/api/v1/llm/chat")
                    .bodyValue(requestBody)
                    .retrieve()
                    .bodyToMono(Map.class)
                    .block();

            String content = extractContent(response);
            return parseLlmResponse(content);
        } catch (Exception e) {
            log.error("Failed to parse intent via LLM", e);
            return List.of();
        }
    }

    private String buildSystemPrompt() {
        var definitions = intentRegistry.findEnabled();
        StringBuilder sb = new StringBuilder();
        sb.append("你是一个意图识别助手。根据用户消息，识别用户意图并提取参数。\n\n");
        sb.append("可用意图列表：\n");
        for (IntentDefinition def : definitions) {
            sb.append("- ").append(def.getIntentId()).append(": ").append(def.getDescription()).append("\n");
            if (def.getSlotsJson() != null) {
                sb.append("  参数: ").append(def.getSlotsJson()).append("\n");
            }
        }
        sb.append("\n请以 JSON 格式返回结果：\n");
        sb.append("{\"intentId\": \"...\", \"confidence\": 0.0-1.0, \"slots\": {...}}\n");
        sb.append("如果无法识别，返回 {\"intentId\": \"unknown\", \"confidence\": 0.0, \"slots\": {}}");
        return sb.toString();
    }

    @SuppressWarnings("unchecked")
    private String extractContent(Map response) {
        try {
            List<Map<String, Object>> choices = (List<Map<String, Object>>) response.get("choices");
            if (choices != null && !choices.isEmpty()) {
                Map<String, Object> message = (Map<String, Object>) choices.get(0).get("message");
                return (String) message.get("content");
            }
        } catch (Exception e) {
            log.error("Failed to extract content from LLM response", e);
        }
        return "";
    }

    private List<ParsedIntent> parseLlmResponse(String content) {
        try {
            // 尝试提取 JSON 块
            String json = content;
            int start = content.indexOf('{');
            int end = content.lastIndexOf('}');
            if (start >= 0 && end > start) {
                json = content.substring(start, end + 1);
            }

            Map<String, Object> parsed = objectMapper.readValue(json, new TypeReference<>() {});
            String intentId = (String) parsed.get("intentId");
            double confidence = parsed.containsKey("confidence")
                    ? ((Number) parsed.get("confidence")).doubleValue() : 0.0;
            Map<String, Object> slots = parsed.containsKey("slots")
                    ? (Map<String, Object>) parsed.get("slots") : Map.of();

            if ("unknown".equals(intentId) || confidence < confidenceThreshold) {
                return List.of();
            }

            return List.of(new ParsedIntent(intentId, confidence, slots));
        } catch (Exception e) {
            log.warn("Failed to parse LLM response as JSON: {}", content, e);
            return List.of();
        }
    }
}

import com.fasterxml.jackson.core.type.TypeReference;
import com.fasterxml.jackson.databind.ObjectMapper;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import org.springframework.data.redis.core.StringRedisTemplate;
import org.springframework.stereotype.Component;
import org.springframework.web.reactive.function.client.WebClient;

import java.util.*;

@Component
public class AiSubstrateContextManager implements ContextManager {

    private static final Logger log = LoggerFactory.getLogger(AiSubstrateContextManager.class);
    private static final String CONTEXT_KEY_PREFIX = "dialogue:ctx:";
    private static final int MAX_CONTEXT_MESSAGES = 20;

    private final StringRedisTemplate redisTemplate;
    private final WebClient aiSubstrateClient;
    private final ObjectMapper objectMapper;

    @Value("${ai-substrate.base-url:http://localhost:8083}")
    private String aiSubstrateBaseUrl;

    public AiSubstrateContextManager(StringRedisTemplate redisTemplate,
                                      WebClient.Builder webClientBuilder,
                                      ObjectMapper objectMapper) {
        this.redisTemplate = redisTemplate;
        this.aiSubstrateClient = webClientBuilder.build();
        this.objectMapper = objectMapper;
    }

    @Override
    public ConversationContext getContext(String conversationId) {
        String key = CONTEXT_KEY_PREFIX + conversationId;
        List<String> messages = redisTemplate.opsForList().range(key, 0, -1);

        List<ConversationContext.ContextMessage> contextMessages = new ArrayList<>();
        if (messages != null) {
            for (String msg : messages) {
                try {
                    Map<String, String> parsed = objectMapper.readValue(msg, new TypeReference<>() {});
                    contextMessages.add(new ConversationContext.ContextMessage(
                            parsed.get("role"), parsed.get("content")));
                } catch (Exception e) {
                    log.warn("Failed to parse context message: {}", msg);
                }
            }
        }

        return new ConversationContext(conversationId, contextMessages, Map.of());
    }

    @Override
    public void appendMessage(String conversationId, String role, String content) {
        String key = CONTEXT_KEY_PREFIX + conversationId;
        try {
            String json = objectMapper.writeValueAsString(Map.of("role", role, "content", content));
            redisTemplate.opsForList().rightPush(key, json);
            // 滑动窗口：保留最近 N 条
            redisTemplate.opsForList().trim(key, -MAX_CONTEXT_MESSAGES, -1);
        } catch (Exception e) {
            log.error("Failed to append context message", e);
        }
    }

    @Override
    public void clearContext(String conversationId) {
        String key = CONTEXT_KEY_PREFIX + conversationId;
        redisTemplate.delete(key);
    }

/**
 * 核心对话管道：
 * 用户消息 → 保存 → NL解析 → 意图路由 → 执行 → 生成回复 → 保存
 */
@Service
public class MessageService {

    private static final Logger log = LoggerFactory.getLogger(MessageService.class);

    private final ConversationRepository conversationRepository;
    private final MessageRepository messageRepository;
    private final NlParser nlParser;
    private final IntentRouter intentRouter;
    private final ResponseGenerator responseGenerator;
    private final ContextManager contextManager;
    private final ObjectMapper objectMapper;

    public MessageService(ConversationRepository conversationRepository,
                          MessageRepository messageRepository,
                          NlParser nlParser,
                          IntentRouter intentRouter,
                          ResponseGenerator responseGenerator,
                          ContextManager contextManager,
                          ObjectMapper objectMapper) {
        this.conversationRepository = conversationRepository;
        this.messageRepository = messageRepository;
        this.nlParser = nlParser;
        this.intentRouter = intentRouter;
        this.responseGenerator = responseGenerator;
        this.contextManager = contextManager;
        this.objectMapper = objectMapper;
    }

    /**
     * 处理用户消息 — 核心管道
     */
    @Transactional
    public Message sendUserMessage(Long conversationId, String userMessage) {
        // 1. 加载会话
        Conversation conversation = conversationRepository.findById(conversationId)
                .orElseThrow(() -> new IllegalArgumentException("会话不存在: " + conversationId));

        if (conversation.getStatus() != ConversationStatus.ACTIVE) {
            throw new IllegalStateException("会话已归档或删除，无法发送消息");
        }

        // 2. 保存用户消息
        Message userMsg = new Message(conversation, MessageRole.USER, userMessage);
        userMsg.setTokenCount(estimateTokens(userMessage));
        messageRepository.save(userMsg);

        // 3. 更新上下文
        contextManager.appendMessage(conversationId.toString(), "user", userMessage);

        // 4. 获取上下文
        ConversationContext context = contextManager.getContext(conversationId.toString());

        // 5. NL 解析
        List<ParsedIntent> intents = nlParser.parse(userMessage, contextToString(context));
        String responseText;

        if (intents.isEmpty()) {
            // 无法识别意图 → fallback
            responseText = responseGenerator.generateFallback(userMessage);
            userMsg.setParsedIntent("{\"intentId\": \"unknown\", \"confidence\": 0.0}");
        } else {
            // 6. 路由 + 执行
            ParsedIntent intent = intents.get(0);
            userMsg.setParsedIntent(toJson(intent));
            IntentHandleResult result = intentRouter.route(intent);
            userMsg.setExecutionResult(toJson(result));

            // 7. 生成回复
            responseText = responseGenerator.generate(intent, result);
        }

        // 8. 保存助手回复
        Message assistantMsg = new Message(conversation, MessageRole.ASSISTANT, responseText);
        assistantMsg.setTokenCount(estimateTokens(responseText));
        messageRepository.save(assistantMsg);

        // 9. 更新上下文
        contextManager.appendMessage(conversationId.toString(), "assistant", responseText);

        log.info("Message processed: conversation={}, userMsg={}, assistantMsg={}",
                conversationId, userMsg.getId(), assistantMsg.getId());

        return assistantMsg;
    }

    private int estimateTokens(String text) {
        // v0.1 粗略估算: 1 token ≈ 4 chars (英文), 1 token ≈ 2 chars (中文)
        return Math.max(1, text.length() / 2);
    }

    private String contextToString(ConversationContext context) {
        return context.recentMessages().stream()
                .map(m -> m.role() + ": " + m.content())
                .reduce("", (a, b) -> a + "\n" + b);
    }

    private String toJson(Object obj) {
        try {
            return objectMapper.writeValueAsString(obj);
        } catch (Exception e) {
            return "{}";
        }
    }
}

package com.metaplatform.dialogue.interfaces.rest;

import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.http.HttpStatus;
import org.springframework.http.ResponseEntity;
import org.springframework.web.bind.annotation.ExceptionHandler;
import org.springframework.web.bind.annotation.RestControllerAdvice;

import java.util.Map;

@RestControllerAdvice
public class GlobalExceptionHandler {

    private static final Logger log = LoggerFactory.getLogger(GlobalExceptionHandler.class);

    @ExceptionHandler(IllegalArgumentException.class)
    public ResponseEntity<Map<String, String>> handleBadRequest(IllegalArgumentException e) {
        return ResponseEntity.badRequest().body(Map.of("error", e.getMessage()));
    }

    @ExceptionHandler(IllegalStateException.class)
    public ResponseEntity<Map<String, String>> handleConflict(IllegalStateException e) {
        return ResponseEntity.status(HttpStatus.CONFLICT).body(Map.of("error", e.getMessage()));
    }

    @ExceptionHandler(Exception.class)
    public ResponseEntity<Map<String, String>> handleGeneric(Exception e) {
        log.error("Unhandled exception", e);
        return ResponseEntity.status(HttpStatus.INTERNAL_SERVER_ERROR)
                .body(Map.of("error", "内部服务错误"));
    }
}
